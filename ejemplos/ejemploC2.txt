func int factorial(int n)
{
  if (n == 0)
  {
    return 1;
  }
  else
  {
    return(n * factorial(n-1));
  }
}

func int iterativeFactorial(int n)
{
    define int fact = 1;
    define int num, i;
    num = n;
    i = num;
    while(i>=1)
    {
       fact=fact*i;
       i = i -1;
    }
    return fact;
}

func main(){
    print(factorial(5));
    print(iterativeFactorial(5));
}
