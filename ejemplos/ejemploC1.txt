
func int suma(int x, int y){
    return x + y ;
}

func int factorial(int x){
    if(x<1){
        return 1;
    }else{
        return x * factorial(x-1);
    }
}

func main(){  
    define int x[10][10];
    define int i, j;
    i=0;
    j=0;
    while(i<10){
        j=0;
        while(j<10){
            x[i][j] = j + i*10;
            j=j+1;
        }
        i=i+1;
    }
}
